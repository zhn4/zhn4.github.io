### 导航栏高亮
注意for循环使用，注意标记元素标记点击元素
```html
<a href="javascript:;" class="btn">1</a>
<a href="javascript:;" class="btn">2</a>
<a href="javascript:;" class="btn">3</a>
<a href="javascript:;" class="btn">4</a>
<a href="javascript:;" class="btn">5</a>

<style>
  .test {
    padding: 10px;
    /*display: block;*/
    border: 1px solid red;
  }
  .active {
    background-color: tomato;
    color: white;
  }
</style>

<script type="text/javascript">
var btn = document.getElementsByClassName('btn');
var num = 0;
for(var i = 0; i < btn.length; i++) {
  btn[i].num = i;//用作标记
  btn[i].addEventListener('click', function() {
    this.classList.add('active');
    btn[num].classList.remove('active');
    num = this.num;
  })
}
</script>
```
